# Ce projet utilise Node.js, pas Python.
# Les dépendances sont gérées par npm via package.json.
#
# Installation :
#
#   Frontend (React + Vite) :
#     cd frontend && npm install
#
#   Backend (Express + MySQL) :
#     cd backend && npm install
#
# Versions Node.js requises : >= 18.0.0
# Base de données           : MySQL >= 8.0
#
# -------------------------------------------------------
# Dépendances backend (backend/package.json)
# -------------------------------------------------------
# express          ^5.2.1   — serveur HTTP
# mysql2           ^3.22.5  — connecteur MySQL
# jsonwebtoken     ^9.0.3   — authentification JWT
# bcryptjs         ^3.0.3   — hashage des mots de passe
# cors             ^2.8.6   — gestion des CORS
# dotenv           ^17.4.2  — variables d'environnement
#
# -------------------------------------------------------
# Dépendances frontend (frontend/package.json)
# -------------------------------------------------------
# react            ^19.2.7  — bibliothèque UI
# react-dom        ^19.2.7  — rendu DOM
# lucide-react     ^1.21.0  — icônes
# recharts         ^3.9.0   — graphiques
# vite             ^8.1.0   — bundler / dev server
